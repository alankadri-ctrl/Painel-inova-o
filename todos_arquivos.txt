# Painel de Projetos · Consigaz

Dashboard interativo para acompanhar prazos e desempenho dos projetos de desenvolvimento da Consigaz.

## 🎯 O que mostra

- **KPIs gerais**: projetos ativos, concluídos, em atraso e cumpridos
- **Totalizador por fase**: tempo médio planejado × real para cada etapa (estruturação, cotação, contrato, implementação)
- **Gantt interativo**: linha do tempo de todos os projetos com filtros e busca
- **Rankings**: maiores atrasos por fase e distribuição por responsável

## 📊 Dados

Base: **236 projetos válidos** extraídos do Excel de controle
- 81 em andamento
- 74 finalizados  
- 37 cancelados
- 44 do Clube de Descontos

Última atualização: Abril 2026

## 🚀 Acesso

➡️ **[Abrir Painel](https://seuusuario.github.io/painel-projetos/)**

function phaseStats(pk){
  const durs_plan = [], durs_real = [];
  let concl = 0, em_and = 0, no_prazo = 0, fora = 0;
  DATA.forEach(p => {
    const f = p.f[pk];
    if(f.dp !== null && f.dp >= 0) durs_plan.push(f.dp);
    if(f.dr !== null && f.dr >= 0) durs_real.push(f.dr);
    if(f.c) concl++;
    if(f.ea) em_and++;
    if(f.sr && f.sr.includes('no prazo')) no_prazo++;
    if(f.sr && f.sr.includes('Fora')) fora++;
  });
  const med = arr => arr.length ? arr.slice().sort((a,b)=>a-b)[Math.floor(arr.length/2)] : null;
  const avg = arr => arr.length ? Math.round(arr.reduce((a,b)=>a+b,0)/arr.length) : null;
  return { avg_plan: avg(durs_plan), avg_real: avg(durs_real), med_plan: med(durs_plan), med_real: med(durs_real), concl, em_and, no_prazo, fora, total_real: durs_real.length };
}

// ======= GANTT =======
let filterSec = 'all';
let searchTerm = '';

function renderGantt(){
  const filtered = DATA.filter(p => {
    if(filterSec !== 'all' && p.s !== filterSec) return false;
    if(searchTerm){
      const t = searchTerm.toLowerCase();
      if(!(p.n||'').toLowerCase().includes(t) && !(p.r||'').toLowerCase().includes(t)) return false;
    }
    // precisa ter pelo menos uma fase com datas
    const temData = Object.keys(PHASE_META).some(pk => {
      const f = p.f[pk];
      return f.i || f.fp || f.r;
    });
    return temData;
  });

  // achar range global
  let minD = null, maxD = null;
  filtered.forEach(p => {
    Object.keys(PHASE_META).forEach(pk => {
      const f = p.f[pk];
      [f.i, f.fp, f.r].forEach(d => {
        const dd = parseD(d);
        if(dd){ if(!minD || dd < minD) minD = dd; if(!maxD || dd > maxD) maxD = dd; }
      });
    });
  });
  if(!minD){ minD = new Date(2022,0,1); maxD = new Date(2026,11,31); }
  // garantir que hoje está incluso
  if(HOJE > maxD) maxD = new Date(HOJE);
  // padding
  minD = new Date(minD.getFullYear(), minD.getMonth(), 1);
  maxD = new Date(maxD.getFullYear(), maxD.getMonth()+2, 1);

  const totalMs = maxD - minD;
  const pct = d => ((d - minD) / totalMs) * 100;

  // ordenar: cancelados primeiro? Não — priorizar por data de aprovação desc (mais recentes em cima)
  // Regra: em andamento primeiro, por data de início de fase atual; depois finalizados; depois resto
  const order = {Andamento:0, Finalizados:1, Clube_Descontos:2, Cancelados:3};
  filtered.sort((a,b) => {
    if(order[a.s] !== order[b.s]) return order[a.s] - order[b.s];
    const da = parseD(a.ap) || new Date(0);
    const db = parseD(b.ap) || new Date(0);
    return db - da;
  });

  document.getElementById('gantt-count').textContent = `${filtered.length} projeto${filtered.length===1?'':'s'}`;

  // Eixo: ticks por trimestre
  const ticks = [];
  let cur = new Date(minD);
  while(cur <= maxD){
    ticks.push(new Date(cur));
    cur.setMonth(cur.getMonth()+3);
  }
  const axisCols = ticks.length;
  const axisHtml = `<div class="gantt-axis" style="grid-template-columns: 280px repeat(${axisCols},1fr)">
    <div style="padding:10px 12px;font-family:'JetBrains Mono',monospace;font-size:10px;color:var(--mute);letter-spacing:.15em;text-transform:uppercase">Projeto · Responsável</div>
    ${ticks.map(t => `<div class="tick">${t.toLocaleDateString('pt-BR',{month:'short'}).replace('.','')} ${String(t.getFullYear()).slice(2)}</div>`).join('')}
  </div>`;

  const todayPct = pct(HOJE);

  const rows = filtered.map((p, idx) => {
    const secDot = {Andamento:'#F59E0B', Finalizados:'#0F766E', Cancelados:'#BE123C', Clube_Descontos:'#7C2D12'}[p.s];
    const bars = [];
    Object.keys(PHASE_META).forEach(pk => {
      const m = PHASE_META[pk];
      const f = p.f[pk];
      const ini = parseD(f.i);
      const fimPrev = parseD(f.fp);
      const real = parseD(f.r);
      // barra planejada (light)
      if(ini && fimPrev && fimPrev > ini){
        const left = pct(ini), right = pct(fimPrev);
        bars.push(`<div class="bar plan" style="left:${left}%;width:${Math.max(right-left,.3)}%;background:${m.color}" data-idx="${idx}" data-ph="${pk}" data-type="plan"></div>`);
      }
      // barra real
      if(ini && real && real >= ini){
        const left = pct(ini), right = pct(real);
        const late = f.sr && f.sr.includes('Fora');
        bars.push(`<div class="bar ${m.key} ${late?'late':''}" style="left:${left}%;width:${Math.max(right-left,.3)}%" data-idx="${idx}" data-ph="${pk}" data-type="real"></div>`);
      }
      // em andamento: barra até hoje
      if(ini && !real && f.ea){
        const left = pct(ini), right = pct(HOJE);
        bars.push(`<div class="bar ${m.key}" style="left:${left}%;width:${Math.max(right-left,.3)}%;opacity:.7" data-idx="${idx}" data-ph="${pk}" data-type="ongoing"></div>`);
      }
    });

    return `<div class="gantt-row">
      <div class="gantt-label" onclick="openModal(${idx})">
        <span class="dotx" style="background:${secDot}"></span>
        <div style="overflow:hidden">
          <div style="overflow:hidden;text-overflow:ellipsis;white-space:nowrap">${escapeHtml(p.n)}</div>
          <div class="resp">${escapeHtml(p.r||'—')}</div>
        </div>
      </div>
      <div class="gantt-track">
        <div class="today" style="left:${todayPct}%" title="Hoje"></div>
        ${bars.join('')}
      </div>
    </div>`;
  }).join('');

  document.getElementById('gantt').innerHTML = axisHtml + rows;
  attachTips(filtered);
}

function escapeHtml(s){ return (s||'').replace(/[&<>"]/g, c => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;'}[c])); }

function attachTips(filtered){
  const tip = document.getElementById('tip');
  document.querySelectorAll('.gantt-track .bar').forEach(el => {
    el.addEventListener('mouseenter', e => {
      const idx = +el.dataset.idx;
      const pk = el.dataset.ph;
      const type = el.dataset.type;
      const p = filtered[idx];
      const f = p.f[pk];
      const m = PHASE_META[pk];
      const typeTxt = type==='plan' ? 'Planejado' : (type==='ongoing' ? 'Em andamento' : 'Execução real');
      const atraso = f.at !== null ? (f.at > 0 ? `<div class="tt-row"><span>Desvio</span><span style="color:#FCA5A5">+${f.at}d</span></div>` : f.at < 0 ? `<div class="tt-row"><span>Desvio</span><span style="color:#86EFAC">${f.at}d</span></div>` : '') : '';
      tip.innerHTML = `
        <div class="tt-name">${escapeHtml(p.n)}</div>
        <div class="tt-row"><span>${m.label}</span><span>${typeTxt}</span></div>
        <div class="tt-row"><span>Início</span><span>${fmtBR(f.i)}</span></div>
        <div class="tt-row"><span>Fim previsto</span><span>${fmtBR(f.fp)}</span></div>
        <div class="tt-row"><span>Real</span><span>${fmtBR(f.r)}</span></div>
        ${f.dp !== null ? `<div class="tt-row"><span>Plan.</span><span>${f.dp}d</span></div>`:''}
        ${f.dr !== null ? `<div class="tt-row"><span>Real</span><span>${f.dr}d</span></div>`:''}
        ${atraso}
      `;
      tip.classList.add('show');
    });
    el.addEventListener('mousemove', e => {
      const tw = tip.offsetWidth, th = tip.offsetHeight;
      let x = e.clientX + 14, y = e.clientY + 14;
      if(x + tw > window.innerWidth - 10) x = e.clientX - tw - 14;
      if(y + th > window.innerHeight - 10) y = e.clientY - th - 14;
      tip.style.left = x+'px'; tip.style.top = y+'px';
    });
    el.addEventListener('mouseleave', () => tip.classList.remove('show'));
    el.addEventListener('click', () => openModal(+el.dataset.idx, filtered));
  });
}

// ======= Modal =======
let currentFiltered = DATA;
function openModal(idx, list){
  const arr = list || currentFiltered;
  const p = arr[idx];
  if(!p) return;
  const secDot = {Andamento:'#F59E0B', Finalizados:'#0F766E', Cancelados:'#BE123C', Clube_Descontos:'#7C2D12'}[p.s];
  const secName = {Andamento:'Em andamento', Finalizados:'Finalizado', Cancelados:'Cancelado', Clube_Descontos:'Clube de Descontos'}[p.s];

  let phaseHtml = '';
  Object.keys(PHASE_META).forEach(pk => {
    const m = PHASE_META[pk];
    const f = p.f[pk];
    if(!f.i && !f.fp && !f.r) return;
    let statCls = 'wait', statTxt = 'Aguardando';
    if(f.sr && f.sr.includes('no prazo')){ statCls='ok'; statTxt='No prazo'; }
    else if(f.sr && f.sr.includes('Fora')){ statCls='late'; statTxt=`Fora do prazo${f.at? ' (+'+f.at+'d)':''}`; }
    else if(f.ea){ statCls='going'; statTxt='Em andamento'; }
    phaseHtml += `
      <div class="ph-item" style="--c:${m.color}">
        <div class="ph-name">${m.label}</div>
        <div class="ph-dates">
          <div><b>Início:</b> ${fmtBR(f.i)} · <b>Fim prev.:</b> ${fmtBR(f.fp)}</div>
          <div><b>Real:</b> ${fmtBR(f.r)}${f.dp !== null ? ` · Plan. ${f.dp}d`:''}${f.dr !== null ? ` · Real ${f.dr}d`:''}</div>
        </div>
        <span class="ph-stat ${statCls}">${statTxt}</span>
      </div>
    `;
  });

  const alerta = p.ma ? `<div class="alert"><b>Motivo do atraso</b>${escapeHtml(p.ma)}${p.pc?`<br><br><b>Parado com</b>${escapeHtml(p.pc)}`:''}</div>` : '';

  document.getElementById('modal-content').innerHTML = `
    <div class="m-tag" style="color:${secDot}">● ${secName}${p.sg?' · '+escapeHtml(p.sg):''}</div>
    <h2>${escapeHtml(p.n)}</h2>
    <div class="m-meta">
      <span><b>Responsável</b>${escapeHtml(p.r||'—')}</span>
      <span><b>Aprovação</b>${fmtBR(p.ap)}</span>
      ${p.ar?`<span><b>Área</b>${escapeHtml(p.ar)}</span>`:''}
      ${p.tp?`<span><b>Tipo</b>${escapeHtml(p.tp)}</span>`:''}
      ${p.pr?`<span><b>Prioridade</b>${escapeHtml(p.pr)}</span>`:''}
    </div>
    <div class="phase-track">${phaseHtml || '<p style="color:var(--mute);font-style:italic">Sem fases preenchidas.</p>'}</div>
    ${alerta}
  `;
  document.getElementById('modal-bg').classList.add('show');
}
function closeModal(){ document.getElementById('modal-bg').classList.remove('show'); }

function renderRanking(){

    Object.keys(PHASE_META).forEach(pk => {

      if(f.at !== null && f.at > 0 && f.sr && f.sr.includes('Fora')){

  fases.sort((a,b) => b.at - a.at);

// ======= Carga por responsável =======
function renderResp(){
  const counts = {};
  DATA.forEach(p => {
    if(p.s === 'Cancelados') return;
    const r = p.r || '—';
    counts[r] = (counts[r]||0)+1;
  });
  const sorted = Object.entries(counts).sort((a,b) => b[1]-a[1]).slice(0,10);
  const max = sorted[0] ? sorted[0][1] : 1;
  document.getElementById('resp-list').innerHTML = sorted.map(([n,c]) => `
    <div class="resp-row">
      <span class="name">${escapeHtml(n)}</span>
      <span class="count">${c}</span>
      <div class="bar-outer"><div class="bar-inner" style="width:${(c/max*100)}%"></div></div>
    </div>
  `).join('');
}
